Encrypt


ROT = 3

def encrypt_ceasar(text):
    plain = ""
    for c in text:
        value = ord(c) + ROT
        if value > 90:
            value = value - 26
        plain += chr(value)
    return plain

>>> print encrypt_ceasar("PYTHONNORDESTE")
SBWKRQQRUGHVWH

Decrypt


ROT = 3

def decrypt_ceasar(text):
    plain = ""
    for c in text:
        value = ord(c) - ROT
        if value < 65:
            value = value + 26
        plain += chr(value)
    return plain

>>> print decrypt_ceasar("SBWKRQQRUGHVWH")
PYTHONNORDESTE
